import { setup, assign } from 'xstate';

export const navMachine = setup({
  types: {
    context: {} as { lang: 'en' | 'pt-br' },
    events: {} as
      | { type: 'GO_HOME' }
      | { type: 'GO_WRITING' }
      | { type: 'GO_ABOUT' }
      | { type: 'GO_CONTACT' }
      | { type: 'SET_LANG'; lang: 'en' | 'pt-br' },
  },
  actions: {
    setLang: assign({
      lang: ({ event }) =>
        event.type === 'SET_LANG' ? event.lang : 'en',
    }),
  },
}).createMachine({
  id: 'nav',
  initial: 'home',
  context: { lang: 'en' },
  on: { SET_LANG: { actions: 'setLang' } },
  states: {
    home:    { on: { GO_WRITING: 'writing',
                     GO_ABOUT:   'about',
                     GO_CONTACT: 'contact' } },
    writing: { on: { GO_HOME:    'home',
                     GO_ABOUT:   'about',
                     GO_CONTACT: 'contact' } },
    about:   { on: { GO_HOME:    'home',
                     GO_WRITING: 'writing',
                     GO_CONTACT: 'contact' } },
    contact: { on: { GO_HOME:    'home',
                     GO_WRITING: 'writing',
                     GO_ABOUT:   'about' } },
  },
});
